021import java.beans.BeanInfo;
022import java.beans.IntrospectionException;
023import java.beans.Introspector;
024import java.beans.PropertyDescriptor;
025import java.beans.PropertyEditor;
026import java.beans.PropertyEditorManager;
027import java.beans.Transient;
028import java.io.StringWriter;
029import java.io.Writer;
030import java.lang.reflect.InvocationTargetException;
031import java.lang.reflect.Method;
032import java.math.BigDecimal;
033import java.math.BigInteger;
034import java.util.Collection;
035import java.util.HashMap;
036import java.util.HashSet;
037import java.util.Map;
038import java.util.Set;

040import javax.json.Json;
041import javax.json.stream.JsonGenerator;

043/**
044 * Encoder for converting a Java object graph to JSON. Objects may be arrays,
045 * collections, maps and JavaBeans.
046 * 
047 * Arrays and collections are converted to JSON arrays. The
048 * type information is lost. Maps and JavaBeans are converted
049 * to JSON objects.
050 * 
051 * The generated JSON objects can have an additional key/value pair with
052 * key "class" and a class name. The class information is generated
053 * only when it is needed, i.e. if it cannot be derived from the containing
054 * object.
055 * 
056 * Given the following classes:
057 *
058 * ```java
059 * public static class Person {
060 *
061 *     private String name;
062 *     private int age;
063 *     private PhoneNumber[] numbers;
064 *
065 *     public String getName() {
066 *         return name;
067 *     }
068 *     
069 *     public void setName(String name) {
070 *         this.name = name;
071 *     }
072 *     
073 *     public int getAge() {
074 *         return age;
075 *     }
076 *     
077 *     public void setAge(int age) {
078 *         this.age = age;
079 *     }
080 *     
081 *     public PhoneNumber[] getNumbers() {
082 *         return numbers;
083 *     }
084 *     
085 *     public void setNumbers(PhoneNumber[] numbers) {
086 *         this.numbers = numbers;
087 *     }
088 * }
089 *
090 * public static class PhoneNumber {
091 *     private String name;
092 *     private String number;
093 *
094 *     public PhoneNumber() {
095 *     }
096 *     
097 *     public String getName() {
098 *         return name;
099 *     }
100 *     
101 *     public void setName(String name) {
102 *         this.name = name;
103 *     }
104 *     
105 *     public String getNumber() {
106 *         return number;
107 *     }
108 *     
109 *     public void setNumber(String number) {
110 *         this.number = number;
111 *     }
112 * }
113 *
114 * public static class SpecialNumber extends PhoneNumber {
115 * }
116 * ```
117 * 
118 * A serialization result may look like this:
119 * 
120 * ```json
121 * {
122 *     "age": 42,
123 *     "name": "Simon Sample",
124 *     "numbers": [
125 *         {
126 *             "name": "Home",
127 *             "number": "06751 51 56 57"
128 *         },
129 *         {
130 *             "class": "test.json.SpecialNumber",
131 *             "name": "Work",
132 *             "number": "030 77 35 44"
133 *         }
134 *     ]
135 * } 
136 * ```
137 * 
138 * 
139 */
140public class JsonBeanEncoder extends JsonCoder {
141
142        private static final Set<String> EXCLUDED_DEFAULT = new HashSet<>();
143
144        static {
145                // See https://issues.apache.org/jira/browse/GROOVY-8284
146                EXCLUDED_DEFAULT.add("groovy.lang.MetaClass");
147        }
148
149        private Map<Class<?>, String> aliases = new HashMap<>();
150        private Set<String> excluded = EXCLUDED_DEFAULT;
151        private JsonGenerator gen;
152        private StringWriter writer = null;
153
154        @Override
155        public JsonBeanEncoder addAlias(Class<?> clazz, String alias) {
156                aliases.put(clazz, alias);
157                return this;
158        }
159
160        /**
161         * Add a type to excude from encoding, usually because it cannot
162         * be converted to JSON. Properties of such types should be
163         * marked as {@link Transient}. However, sometimes base types
164         * don't follow the rules.
165         * 
166         * @param className
167         * @return the encoder for easy chaining
168         */
169        public JsonBeanEncoder addExcluded(String className) {
170                if (excluded == EXCLUDED_DEFAULT) {
171                        excluded = new HashSet<>(EXCLUDED_DEFAULT);
172                }
173                excluded.add(className);
174                return this;
175        }
176
177        /**
178         * Create a new encoder using a default {@link JsonGenerator}. 
179         * 
180         * @param out the sink
181         * @return the encoder
182         */
183        public static JsonBeanEncoder create(Writer out) {
184                return new JsonBeanEncoder(Json.createGenerator(out));
185        }
186
187        /**
188         * Create a new encoder using a default {@link JsonGenerator}
189         * that writes to an internally created {@link StringWriter}. 
190         * The result can be obtained by invoking {@link #toJson()}.
191         * 
192         * @return the encoder
193         */
194        public static JsonBeanEncoder create() {
195                return new JsonBeanEncoder();
196        }
197
198        /**
199         * Create a new encoder using the given {@link JsonGenerator}. 
200         * 
201         * @param generator the generator
202         * @return the encoder
203         */
204        public static JsonBeanEncoder create(JsonGenerator generator) {
205                return new JsonBeanEncoder(generator);
206        }
207
208        private JsonBeanEncoder() {
209                writer = new StringWriter();
210                gen = Json.createGenerator(writer);
211        }
212
213        private JsonBeanEncoder(JsonGenerator generator) {
214                gen = generator;
215        }
216
217        /**
218         * Returns the text written to the output. Can only be used
219         * if the encoder has been created with {@link #JsonBeanEncoder()}.
220         * 
221         * @return the result
222         */
223        public String toJson() {
224                if (writer == null) {
225                        throw new IllegalStateException(
226                                        "JsonBeanEncoder has been created without a known writer.");
227                }
228                return writer.toString();
229        }
230        
231        public JsonBeanEncoder writeArray(Object... items) {
232                doWriteObject(items, items.getClass());
233                gen.flush();
234                return this;
235        }
236        
237        public JsonBeanEncoder writeObject(Object obj) {
238                doWriteObject(obj, obj.getClass());
239                gen.flush();
240                return this;
241        }
242        
243        private void doWriteObject(Object obj, Class<?> expectedType) {
244                if (obj == null) {
245                        gen.writeNull();
246                        return;
247                }
248                if (obj instanceof Boolean) {
249                        gen.write((Boolean)obj);
250                        return;
251                } 
252                if (obj instanceof Byte) {
253                        gen.write(((Byte)obj).intValue());
254                        return;
255                } 
256                if (obj instanceof Number) {
257                        if (obj instanceof Short) {
258                                gen.write((Short)obj);
259                                return;
260                        }
261                        if (obj instanceof Integer) {
262                                gen.write((Integer)obj);
263                                return;
264                        }
265                        if (obj instanceof Long) {
266                                gen.write((Long)obj);
267                                return;
268                        }
269                        if (obj instanceof BigInteger) {
270                                gen.write((BigInteger)obj);
271                                return;
272                        }
273                        if (obj instanceof BigDecimal) {
274                                gen.write((BigDecimal)obj);
275                                return;
276                        }
277                        gen.write((Double)obj);
278                        return;
279                }
280                if (obj instanceof Object[]) {
281                        gen.writeStartArray();
282                        Class<?> compType = null;
283                        if (expectedType != null && expectedType.isArray()) {
284                                compType = expectedType.getComponentType();
285                        }
286                        for (Object item: (Object[])obj) {
287                                doWriteObject(item, compType);
288                        }
289                        gen.writeEnd();
290                        return;
291                }
292                if (obj instanceof Collection) {
293                        gen.writeStartArray();
294                        for (Object item: (Collection<?>)obj) {
295                                doWriteObject(item, null);
296                        }
297                        gen.writeEnd();
298                        return;
299                }
300                if (obj instanceof Map) {
301                        @SuppressWarnings("unchecked")
302                        Map<String,Object> map = (Map<String,Object>)obj;
303                        gen.writeStartObject();
304                        for (Map.Entry<String, Object> e: map.entrySet()) {
305                                gen.writeKey(e.getKey());
306                                doWriteObject(e.getValue(), null);
307                        }
308                        gen.writeEnd();
309                        return;
310                }
311                PropertyEditor propertyEditor 
312                        = PropertyEditorManager.findEditor(obj.getClass());
313                if (propertyEditor != null) {
314                        propertyEditor.setValue(obj);
315                        gen.write(propertyEditor.getAsText());
316                        return;
317                }
318                try {
319                        BeanInfo beanInfo = Introspector.getBeanInfo(
320                                        obj.getClass(), Object.class);
321                        if (beanInfo.getPropertyDescriptors().length > 0) {
322                                gen.writeStartObject();
323                                if (!obj.getClass().equals(expectedType)) {
324                                        gen.write("class", aliases.computeIfAbsent(
325                                                        obj.getClass(), k -> k.getName()));
326                                }
327                                for (PropertyDescriptor propDesc: 
328                                                beanInfo.getPropertyDescriptors()) {
329                                        if (propDesc.getValue("transient") != null) {
330                                                continue;
331                                        }
332                                        if (excluded.contains(propDesc.getPropertyType().getName())) {
333                                                continue;
334                                        }
335                                        Method method = propDesc.getReadMethod();
336                                        if (method == null) {
337                                                continue;
338                                        }
339                                        try {
340                                                Object value = method.invoke(obj);
341                                                gen.writeKey(propDesc.getName());
342                                                doWriteObject(value, propDesc.getPropertyType());
343                                                continue;
344                                        } catch (IllegalAccessException | IllegalArgumentException
345                                                | InvocationTargetException e) {
346                                                // Bad luck
347                                        }
348                                }
349                                gen.writeEnd();
350                                return;
351                        }
352                } catch (IntrospectionException e) {
353                        // No luck
354                }
355                // Last resort
356                gen.write(obj.toString());
357        }
