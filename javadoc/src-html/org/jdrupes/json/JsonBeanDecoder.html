043/**
044 * Decoder for converting JSON to a Java object graph. The decoding
045 * is based on the expected type passed to the decode method.
046 * 
047 * The convertion rules are as follows:
048 *  * If the expected type is {@link Object} and the JSON input
049 *    is an array, the JSON array is
050 *    converted to an {@link ArrayList} with element type {@link Object}.
051 *  * If the expected type implements {@link Collection} 
052 *    a container of the expected type is created. Its element type
053 *    is also {@link Object}. The JSON input must be an array.
054 *  * If the expected type is an array type, an array of the expected 
055 *    type with the given element type is created. The JSON input must
056 *    be an array.
057 *  * In all cases above, the element type is passed 
058 *    as expected type when decoding the members of the JSON array.
059 *  * If the expected type is an {@link Object} and the JSON input
060 *    is a JSON object, the input is converted to a {@link HashMap}.
061 *  * If the expected type is neither of the above, it is assumed
062 *    to be a JavaBean and the JSON input must be a JSON object.
063 *    The key/value pairs of the JSON input are interpreted as properties
064 *    of the JavaBean and set if the values have successfully been parsed.
065 *    The type of the properties are passed as expected types when
066 *    parsing the values.
067 *      
068 *  A JSON object can have a "class" key. It must be the first key
069 *  of the object. Its value is used to instantiate the Java object
070 *  in which the information of the JSON object is stored. If
071 *  provided, the class specified by this key/value pair overrides 
072 *  the class passed as expected class. It is checked, however, that the
073 *  specified class is assignable to the expected class.
074 *  
075 *  The value specified is first matched against the aliases that
076 *  have been registered with the decoder 
077 *  (see {@link #addAlias(Class, String)}). If no match is found,
078 *  the converter set with {@link JsonBeanDecoder#setClassConverter(Function)}
079 *  is used to convert the name to a class. The function defaults
080 *  to {@link Class#forName(String)}. If the converter does not
081 *  return a result, a {@link HashMap} is used as container for 
082 *  the values provided by the JSON object.
083 */
084public class JsonBeanDecoder extends JsonCoder {
085
086        private Map<String,Class<?>> aliases = new HashMap<>();
087        private Function<String,Optional<Class<?>>> classConverter 
088                = name -> {
089                        try {
090                                return Optional.ofNullable(Class.forName(name));
091                        } catch (ClassNotFoundException e) {
092                                return Optional.empty();
093                        }
094                };
095        private JsonParser parser;
096        
097        @Override
098        public JsonBeanDecoder addAlias(Class<?> clazz, String alias) {
099                aliases.put(alias, clazz);
100                return this;
101        }
102
103        /**
104         * Sets the converter that maps a specified "class" to an actual Java
105         * {@link Class}. If it does not return a class, a {@link HashMap} is 
106         * used to store the data of the JSON object. 
107         * 
108         * @param converter the converter to use
109         * @return the conversion result
110         */
111        public JsonBeanDecoder setClassConverter(
112                        Function<String,Optional<Class<?>>> converter) {
113                this.classConverter = converter;
114                return this;
115        }

124                return new JsonBeanDecoder(Json.createParser(in));

149        public JsonBeanDecoder(JsonParser parser) {
150                this.parser = parser;
151        }
152
153        public Map<String,?> readObject() {
154                Map <String,?> root = new HashMap<>();
155                return root;
156        }
157
158        public <T> T readObject(Class<T> expected) throws JsonDecodeException {
159                return readValue(expected);
160        }
161        
162        public <T> T readArray(Class<T> expected) throws JsonDecodeException {
163                return readValue(expected);
164        }
165        
166        @SuppressWarnings("unchecked")
167        private <T> T readValue(Class<T> expected) 
168                        throws JsonDecodeException {
169                switch(parser.next()) {
170                case END_ARRAY:
171                case END_OBJECT:
172                        return null;
173                case VALUE_NULL:
174                        return null;
175                case VALUE_FALSE:
176                        return (T)Boolean.FALSE;
177                case VALUE_TRUE:
178                        return (T)Boolean.TRUE;
179                case VALUE_STRING:
180                case KEY_NAME:
181                        return (T)parser.getString();
182                case VALUE_NUMBER:
183                        if (expected.equals(Byte.class) 
184                                        || expected.equals(Byte.TYPE)) {
185                                return (T)Byte.valueOf((byte)parser.getInt());
186                        }
187                        if (expected.equals(Short.class) 
188                                        || expected.equals(Short.TYPE)) {
189                                return (T)Short.valueOf((short)parser.getInt());
190                        }
191                        if (expected.equals(Integer.class) 
192                                        || expected.equals(Integer.TYPE)) {
193                                return (T)Integer.valueOf(parser.getInt());
194                        }
195                        if (expected.equals(BigInteger.class)) {
196                                return (T)parser.getBigDecimal().toBigInteger();
197                        }
198                        if (expected.equals(BigDecimal.class)) {
199                                return (T)parser.getBigDecimal();
200                        }
201                        if (expected.equals(Float.class)
202                                        || expected.equals(Float.TYPE)) {
203                                return (T)Float.valueOf(parser.getBigDecimal().floatValue());
204                        }
205                        if (expected.equals(Long.class) 
206                                        || parser.equals(Long.TYPE)
207                                        || parser.isIntegralNumber()) {
208                                return (T)Long.valueOf(parser.getLong());
209                        }
210                        return (T)Double.valueOf(parser.getBigDecimal().doubleValue());
211                case START_ARRAY:
212                        if (expected.isArray() 
213                                        || Collection.class.isAssignableFrom(expected)
214                                        || expected.equals(Object.class)) {
215                                return (T)readArrayValue(expected);
216                        }
217                        throw new JsonDecodeException(parser.getLocation()
218                                        + ": Encountered unexpected array.");
219                case START_OBJECT:
220                        return readObjectValue(expected);
221                }
222                throw new JsonDecodeException(parser.getLocation()
223                                + ": Unexpected event.");
224        }
225        
226        @SuppressWarnings("unchecked")
227        private <T> Object readArrayValue(Class<T> arrayType) 
228                        throws JsonDecodeException {
229                Collection<T> items;
230                if (Collections.class.isAssignableFrom(arrayType)
231                                && !arrayType.isInterface()) {
232                        try {
233                                items = (Collection<T>)arrayType.newInstance();
234                        } catch (InstantiationException | IllegalAccessException e) {
235                                throw new JsonDecodeException(parser.getLocation()
236                                                + ": Cannot create " + arrayType.getName(), e);
237                        }
238                } else {
239                        items = new ArrayList<>();
240                }
241                Class<?> itemType = Object.class;
242                if (arrayType.isArray()) {
243                        itemType = arrayType.getComponentType();
244                }
245                while (true) {
246                        T item = (T)readValue(itemType);
247                        if (item == null) {
248                                break;
249                        }
250                        items.add(item);
251                }
252                if (!arrayType.isArray()) {
253                        return items;
254                }
255                Object result = Array.newInstance(itemType, items.size());
256                int index = 0;
257                for (Object o: items) {
258                        Array.set(result, index++, o);
259                }
260                return result;
261        }
262        
263        private <T> T readObjectValue(Class<T> expected) 
264                        throws JsonDecodeException {
265                Event event = parser.next();
266                if (!event.equals(Event.KEY_NAME)) {
267                        throw new JsonDecodeException(parser.getLocation()
268                                        + ": Unexpected Json event " + event);
269                }
270                String key = parser.getString();                
271                Class<?> cls = expected;
272                if ("class".equals(key)) {
273                        parser.next();
274                        String provided = parser.getString();
275                        if (aliases.containsKey(provided)) {
276                                cls = aliases.get(provided);
277                        } else {
278                                cls = classConverter.apply(provided).orElse(HashMap.class);
279                        }
280                }
281                if (!expected.isAssignableFrom(cls)) {
282                        throw new JsonDecodeException(parser.getLocation()
283                                        + ": Expected " + expected.getName()
284                                        + " found " + cls.getName());
285                }
286                if (cls.equals(Object.class)) {
287                        cls = HashMap.class;
288                }
289                T result = null;
290                try {
291                        @SuppressWarnings("unchecked")
292                        T res = (T)cls.newInstance();
293                        result = res;
294                } catch (InstantiationException
295                                | IllegalAccessException e) {
296                        throw new JsonDecodeException(parser.getLocation()
297                                        + ": Cannot create " + cls.getName(), e);
298                }
299                if (result instanceof Map) {
300                        @SuppressWarnings("unchecked")
301                        Map<String,?> map = (Map<String,?>)result;
302                        @SuppressWarnings("unchecked")
303                        T res = (T)objectToMap(map, !"class".equals(key));
304                        return res;
305                }
306                return objectToBean(result, !"class".equals(key));
307        }
308
309        private Map<String,?> objectToMap(Map<String,?> result, boolean inKeyState)
310                        throws JsonDecodeException {
311                whileLoop:
312                while (true) {
313                        Event event = inKeyState ? Event.KEY_NAME : parser.next();
314                        switch(event) {
315                        case END_OBJECT:
316                                break whileLoop;
317                                
318                        case KEY_NAME:
319                                String key = parser.getString();
320                                inKeyState = false;
321                                Object value = readValue(Object.class);
322                                @SuppressWarnings("unchecked")
323                                Map<String,Object> map = (Map<String,Object>)result;
324                                map.put(key, value);
325                                break;
326                                
327                        default:
328                                throw new JsonDecodeException(parser.getLocation()
329                                                + ": Unexpected Json event " + event);
330                        }
331                }
332                return result;
333                
334        }
335        
336        private <T> T objectToBean(T result, boolean inKeyState)
337                        throws JsonDecodeException {
338                Map<String,PropertyDescriptor> beanProps = new HashMap<>();
339                try {
340                        BeanInfo beanInfo = Introspector.getBeanInfo(
341                                        result.getClass(), Object.class);
342                        for (PropertyDescriptor 
343                                        p: beanInfo.getPropertyDescriptors()) {
344                                beanProps.put(p.getName(), p);
345                        }
346                } catch (IntrospectionException e) {
347                        throw new JsonDecodeException(parser.getLocation()
348                                        + ": Cannot introspect " + result.getClass());
349                }
350                
351                whileLoop:
352                while (true) {
353                        Event event = inKeyState ? Event.KEY_NAME : parser.next();
354                        switch(event) {
355                        case END_OBJECT:
356                                break whileLoop;
357                                
358                        case KEY_NAME:
359                                String key = parser.getString();
360                                inKeyState = false;
361                                PropertyDescriptor property = beanProps.get(key);
362                                if (property == null) {
363                                        new JsonDecodeException(parser.getLocation()
364                                                        + ": No bean property for key " + key);
365                                }
366                                Object value = readValue(property.getPropertyType());
367                                try {
368                                        property.getWriteMethod().invoke(result, value);
369                                } catch (IllegalAccessException | IllegalArgumentException
370                                        | InvocationTargetException e) {
371                                        throw new JsonDecodeException(parser.getLocation()
372                                                        + ": Cannot write property " + key);
373                                }
374                                break;
375                                
376                        default:
377                                throw new JsonDecodeException(parser.getLocation()
378                                                + ": Unexpected Json event " + event);
379                        }
380                }
381                return result;
382                
383        }
